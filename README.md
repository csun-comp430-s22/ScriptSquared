# ScriptSquared

## Syntax

- var is a variable
- s is a String
- i is an Int
- b is a Boolean
- methodname is the name of a method
- classname is the name of the class


type ::= Int | String | Boolean | Void | classname

op ::= + | - | * | / | > | < | == | != | >= | <=

exp ::= i | s | b | this | exp op exp | exp.methodname(exp*) | new classname(exp*)
